import { inject } from '@angular/core'; import { CanActivateFn, Router } from '@angular/router'; import { IamService } from '../../iam/application/iam.service'; /** * Cuando conectes la ruta de login en `app.routes.ts`, añade `canActivate: [authGuard]` al layout * y registra `path: 'login'` con `LoginPage`. */ export const authGuard: CanActivateFn = () => { const auth = inject(IamService); const router = inject(Router); if (auth.isAuthenticated()) { return true; } return router.createUrlTree(['/login']); };
